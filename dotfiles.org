  # Argument should be one of: h j k l

  case "$key" in
    h) dir="left" ;;
    j) dir="down" ;;
    k) dir="up" ;;

    ,*) exit 1 ;;
  esac

We only want to change the post-command in the config for waypaper and don't really care about any other option. The post-command setting is injected into the config file using =sed=. We also make sure the config exists, and if not create a minimal one.
#+begin_src emacs-lisp
  (defun hesam-waypaper--ensure-config-exists ()
    "Ensure the waypaper config directory and file exist with a header."
    (let* ((path (expand-file-name "waypaper/config.ini" (xdg-config-home)))
           (dir (file-name-directory path)))
      (unless (file-directory-p dir)
        (make-directory dir t))
      (unless (file-exists-p path)
        (write-region "[Settings]\n" nil path))))

  (defun hesam-waypaper--inject-postcommand ()
    "Surgically update or append the post_command in config.ini."
    (let* ((path (expand-file-name "waypaper/config.ini" (xdg-config-home)))
           (cmd "emacsclient --eval \"(hesam-matugen-apply nil \\\"$wallpaper\\\")\"")
           (line (format "post_command = %s" cmd)))
      (shell-command
       (format "grep -q '^post_command =' %s && sed -i 's|^post_command =.*|%s|' %s || echo '%s' >> %s"
               path
               (shell-quote-argument line)
               path
               line
               path))))

  (hesam-waypaper--ensure-config-exists)
  (hesam-waypaper--inject-postcommand)

** external backlight
Control backlight of external monitors with =ddcutil=.
#+begin_src conf :tangle ~/.local/bin/external-backlight.sh :shebang "#!/bin/bash"
  # setvcp arguments are + N to increase N or - N to decrease
  # $* joins all arguments (e.g. + and 10) into a single string
  ddcutil setvcp 10 $* --noverify

  # Get the new value to show in the OSD
  # --terse format: VCP 10 C 50 100 (where 50 is current, 100 is max)
  current=$(ddcutil getvcp 10 --terse | awk '{print $4}')

  # Trigger SwayOSD to show the bar
  swayosd-client --brightness "$current"
#+end_src
